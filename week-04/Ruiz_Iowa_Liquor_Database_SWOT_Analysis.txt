-- 2. Which transactions for your [Category/Vendor] had a bottle quantity greater than 12?
-- Display the date, store number, item description, and total amount. 
-- (Strength: Identifying bulk buyers or wholesale-style transactions).

SELECT sales.date, stores.store, sales.description
FROM sales
JOIN stores ON sales.store = stores.store
WHERE bottle_qty > 12 AND vendor ILIKE '%Bacar%'


-- 3. Find all products in the products_table whose item_description contains a specific
-- keyword (e.g., 'Limited', 'Spiced'). What categories do they belong to?
-- (Opportunity: Identifying niche product variants).

SELECT category_name, item_description, vendor_name
FROM products
WHERE (item_description ILIKE '%Limi%'
	OR item_description ILIKE '%Spi%'
	OR item_description ILIKE '%Yr%'
	OR item_description ILIKE '%Reser%'
	OR item_description ILIKE '%Orig%'
	OR item_description ILIKE '%Exclusi%'
	OR item_description ILIKE '%Spec%')
	AND vendor_name ILIKE '%Bacar%'
GROUP BY category_name, item_description, vendor_name
Order By 3 ASC

-- 5. How many transactions were recorded for each specific item description within your 
-- chosen [Category]? Which specific product is the most frequently purchased? 
-- (Strength: Identifying your "hero" product).
-- Bacardi Superior Rum (28,786), Grey Goose Vodka (24,697), and Bacardi Limon (11,303) are the top selling products for Bacardi
SELECT COUNT (*), description, vendor
FROM sales
WHERE vendor ILIKE '%Bacar%'
GROUP BY description, vendor
	ORDER BY count DESC

-- 6. Which store generated the highest total revenue for your [Category/Vendor]? 2633 "Hy-vee #3 / Bdi / Des Moines", $983,828.66
-- Which generated the lowest (but still greater than zero)? 4875, "Bani's", $14.22
-- (Strength vs. Weakness: Identifying your best and worst retail partners). 

SELECT sales.store, stores.name, SUM(sales.total)::money AS Total_Sales_Revenue
FROM sales
JOIN stores ON sales.store = stores.store
WHERE  vendor ILIKE '%Bacar%' 
GROUP BY sales.store, stores.name
HAVING SUM(sales.total) > 0
ORDER BY Total_Sales_Revenue DESC -- ASC to find the lowest

SELECT CASE

	ELSE vendor
	END AS new_vendor, 
SUM(sales.total)::money AS Total_Sales_Revenue
FROM sales

	CASE	
		WHEN vendor ILIKE '%Bacar%'THEN 'Bacardi'
		ELSE vendor
	END

ORDER BY Total_Sales_Revenue DESC; -- ASC to find the lowest

-- 8. Which stores had total sales revenue for your [Category/Vendor] exceeding $2,000? 930 total stores
-- (Hint: Use HAVING to filter aggregated store totals). 
-- (Strength: Pinpointing high-performing retail locations).

SELECT sales.store, stores.name, SUM(sales.total)::money AS Total_Sales_Revenue
FROM sales
JOIN stores ON sales.store = stores.store
WHERE  vendor ILIKE '%Bacar%' 
GROUP BY sales.store, stores.name
HAVING SUM(sales.total) > 2000
ORDER BY Total_Sales_Revenue DESC

-- 10. List all transactions where the state bottle cost was between $10 and $20 for 
-- your [Category/Vendor]. -- (Opportunity: Analyzing performance in the "mid-tier" price bracket).

	WHEN vendor ILIKE '%Bacar%'THEN 'Bacardi'
	ELSE vendor
	END AS new_vendor, state_btl_cost, total
FROM sales
WHERE vendor ILIKE '%Bacar%' --THEN 'Bacardi'
AND state_btl_cost::numeric BETWEEN 10 AND 20
ORDER BY 1 DESC; -- ASC to find the lowest
