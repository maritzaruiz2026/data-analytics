-- 14. Join the sales_table and products_table to show total revenue for your [Vendor] 
-- alongside the proof and pack size of the items. 
-- (Strength: Determining if higher proof or larger packs drive more revenue). 
SELECT sales.item, 
TRIM(REGEXP_REPLACE(REGEXP_REPLACE(products.item_description, '''', ''), 
'\s+', ' ')) AS clean_description, SUM(sales.total)::money AS Total_Sales_Revenue_For_Bacardi, products.proof, products.pack, products.inner_pack
FROM sales
JOIN products ON sales.item = products.item_no
WHERE products.vendor_name ILIKE '%Bacar%' 
GROUP BY sales.item, TRIM(
        REGEXP_REPLACE(
            REGEXP_REPLACE(products.item_description, '''', ''),
        '\s+', ' ')
    ), sales.description, products.proof, products.pack, products.inner_pack
ORDER BY sales.item ASC;
